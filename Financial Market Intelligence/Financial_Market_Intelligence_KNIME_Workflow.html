🔄 7-Day Iterative Prediction (Python Script Node)
# Python Script Node — 7 Day Rolling Prediction

import joblib
import pandas as pd
import numpy as np
from datetime import timedelta

# Input: input_table (from KNIME)
df = input_table.copy()
gold_model = joblib.load("models/gold_model.pkl")

future_preds = []
future_dates = []

for i in range(7):
lag1 = df['Gold_24K_1g'].iloc[-1]
lag2 = df['Gold_24K_1g'].iloc[-2]
lag3 = df['Gold_24K_1g'].iloc[-3]
ma7 = df['Gold_24K_1g'].tail(7).mean()
ma30 = df['Gold_24K_1g'].tail(30).mean()
usd = df['USD_INR'].iloc[-1]

X = pd.DataFrame([[lag1, lag2, lag3, ma7, ma30, usd, 0,
lag1*(22/24), lag1*(18/24), 18/24/lag1, 0]],
columns=['Lag_1','Lag_2','Lag_3','MA_7','MA_30',
'USD_INR','USD_Change','Gold_22K_1g','Gold_18K_1g',
'Gold_18K_Ratio','DayOfWeek'])

pred = gold_model.predict(X)[0]
future_preds.append(pred)
next_date = df['Date'].iloc[-1] + timedelta(days=1)
future_dates.append(next_date)

new_row = df.iloc[-1].copy()
new_row['Date'] = next_date
new_row['Gold_24K_1g'] = pred
df = pd.concat([df, pd.DataFrame([new_row])], ignore_index=True)

output_table = pd.DataFrame({
'Date': future_dates,
'Gold_24K_Predicted': future_preds
})
